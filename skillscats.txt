import React from 'react';

const skillsData = {
  categories: [
    {
      category: "Languages", 
      subcategories: ["JavaScript", "TypeScript", "Python"]
    },
    {
      category: "IT Security",
      subcategories: ["SSO integrations", "2FA management", "Vulnerability scanning and remediation"]
    }
  ]
}

function Skills() {
  return (
    <div className="skills">
      {skillsData.categories.map(category => (
        <SkillCategory 
          key={category.category}
          category={category.category} 
          subcategories={category.subcategories}
        />  
      ))}
    </div>
  )
}

function SkillCategory({ category, subcategories }) {
  return (
    <div className="skill-category">
      <h3>{category}</h3>
      <ul>
        {subcategories.map(subcategory => (
          <li key={subcategory}>{subcategory}</li>
        ))}  
      </ul>
    </div>
  )
}

export default Skills;

.skills {
  display: flex;
  flex-direction: column;
}

.skill-category {
  margin-bottom: 20px;
}

.skill-category h3 {
  margin-bottom: 10px;  
}

.skill-category ul {
  list-style-type: circle;
  padding-left: 20px;
}
